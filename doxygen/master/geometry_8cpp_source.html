83 bool lineIntersection(
84  const Eigen::Ref<const Eigen::Vector2d> &a,
85  const Eigen::Ref<const Eigen::Vector2d> &b,
86  const Eigen::Ref<const Eigen::Vector2d> &c,
87  const Eigen::Ref<const Eigen::Vector2d> &d,
88  Eigen::Ref<Eigen::Vector2d> & intersectionPoint)
89 {
90  // Compute denominator for solving 2x2 equation system
91  double D = a(0) * (d(1) - c(1)) +
92  b(0) * (c(1) - d(1)) +
93  d(0) * (b(1) - a(1)) -
94  c(0) * (a(1) - b(1));
95 
96  // If D==0, the two lines are parallel
97  if (math::equals(D, 0.0)) {
98  return false;
99  }
100 
101  // Compute line segment parameter s, which is in [0, 1], if the
102  // intersection point is within the segment.
103  double s = (a(0) * (d(1) - c(1)) + c(0) * (a(1) - d(1)) + d(0) * (c(1) - a(1))) / D;
104 
105  intersectionPoint = a + s * (b - a);
106  return true;
107 }

109 ResultConstants segmentPlaneIntersection(
110  const Eigen::Vector3d &pointOnPlane,
111  const Eigen::Vector3d &planeNormal,
112  const Eigen::Vector3d &firstPointSegment,
113  const Eigen::Vector3d &secondPointSegment,
114  Eigen::Vector3d & intersectionPoint)
115 {
116  // Methodology of "Computation Geometry in C", Joseph O'Rourke, Chapter 7.3.1
117 
118  // The plane is represented by: p(x,y,z) dot planeNormal = d (1)
119  // We need to compute d by using pointOnPlane as p(x,y,z):
120  double d = pointOnPlane.dot(planeNormal);
121 
122  // The segment is represented by:
123  // firstPointSegment + (secondPointSegment - firstPointSegment) * t (2)
124  // We insert (2) into (1) and reorder for t. The resulting equations reads
125  // t = (d - firstPointSegment dot planeNormal) /
126  // ((secondPointSegment - firstPointSegment) dot planeNormal). (3)
127  //
128  // If the denominator of (3) equals 0, the segment is parallel to the plane.
129  // If, in addition, the denominator of (3) equals 0, the segment lies in the
130  // plane. Otherwise we can compute a point of intersection.
131  Eigen::Vector3d segmentVec(secondPointSegment - firstPointSegment);
132  double nominator = d - firstPointSegment.dot(planeNormal);
133  double denominator = segmentVec.dot(planeNormal);
134  if (math::equals(denominator, 0.0)) {
135  if (math::equals(nominator, 0.0)) {
136  return CONTAINED;
137  } else {
138  return NO_INTERSECTION;
139  }
140  }
141  double t = nominator / denominator;
142 
143  // If t is larger than 1 or smaller than zero, the intersection is not within
144  // the line segment
145  if (math::greater(t, 1.0) || math::greater(0.0, t)) {
146  return NO_INTERSECTION;
147  }
148 
149  intersectionPoint = firstPointSegment + segmentVec * t;
150 
151  // If t equals 1 or 0, the segment is just touching the plane, otherwise, a
152  // real intersection is happening.
153  if (math::equals(t, 0.0) || math::equals(t, 1.0)) {
154  return TOUCHING;
155  }
156 
157  return INTERSECTION;
158 }

251 ConvexityResult isConvexQuad(std::array<Eigen::VectorXd, 4> coords)
252 {
253  /*
254  All points need to be projected into a new plane with only 2 coordinates, x' and y'. These are used to check
255  the convexity of the quad. These new coordinates are stored in 'coords'.
256  */
257  PRECICE_ASSERT(std::all_of(coords.cbegin(), coords.cend(),
258  [](const auto &v) { return v.size() == 3; }),
259  "This only works in 3D.");
260 
261  Eigen::Vector3d coordOrigin; // Origin point for the transformation of points onto the new plane
262  coordOrigin = coords[0];
263 
264  // Normal of the plane of first three points in the list of vertices
265  Eigen::Vector3d e_1 = coords[1] - coordOrigin;
266  Eigen::Vector3d e_2 = coords[2] - coordOrigin;
267  Eigen::Vector3d normalVector = e_1.cross(e_2);
268 
269  PRECICE_CHECK(math::equals(normalVector.dot(coords[3] - coordOrigin), 0.0),
270  "Non-planar quads are not supported. The vertex coordinates are: {}.", coords);
271 
272  //Transform Coordinates - coord[0] is the origin
273  for (int i = 0; i < 4; i++) {
274  Eigen::Vector3d coordinateDifference = coords[i] - coordOrigin;
275  coords[i][0] = e_1.dot(coordinateDifference);
276  coords[i][1] = e_2.dot(coordinateDifference);
277  coords[i][2] = normalVector.dot(coordinateDifference);
278  }
279 
280  /*
281  For the convex hull algorithm, the most left hand point regarding the x coordinate is chosen as the starting point.
282  The algorithm moves in an anti-clockwise position, finding the most right hand coordinate from the
283  previous most right hand point. The algorithm must find 3 other points in order to be a valid quad.
284  */
285 
286  //First find point with smallest x coord. This point must be in the convex set then and is the starting point of gift wrapping algorithm
287  int idLowestPoint = 0;
288  for (int i = 1; i < 4; i++) {
289  if (coords[i][0] < coords[idLowestPoint][0]) {
290  idLowestPoint = i;
291  }
292  }
293 
294  // Found starting point. Add this as the first vertex in the convex hull.
295  // current is the origin point => hull[0]
296  int validVertexIDCounter = 0; // Counts number of times a valid vertex is found. Must be 4 for a valid quad.
297  int currentVertex = idLowestPoint; // current valid vertex
298  ConvexityResult result{};
299  do {
300  // Add current point to result
301  result.vertexOrder[validVertexIDCounter] = currentVertex;
302 
303  // Next potential valid vertex
304  int nextVertex = (currentVertex + 1) % 4; // remainder resets loop through vector of points
305  for (int i = 0; i < 4; i++) {
306  double y1 = coords[currentVertex][1] - coords[nextVertex][1];
307  double y2 = coords[currentVertex][1] - coords[i][1];
308  double x1 = coords[currentVertex][0] - coords[nextVertex][0];
309  double x2 = coords[currentVertex][0] - coords[i][0];
310  double val = y2 * x1 - y1 * x2;
311 
312  if (val > 0) {
313  nextVertex = i;
314  }
315  }
316  // Now nextVertex is the most anti-clockwise with respect to current
317  // Set current as nextVertex for next iteration, so that nextVertex is added to
318  // result 'hull'
319  currentVertex = nextVertex;
320  validVertexIDCounter++;
321  } while (currentVertex != idLowestPoint); // While we don't come to first point
322 
323  //Ordering of quad is hull 0-1-2-3-0
324  result.convex = (validVertexIDCounter == 4);
325 
326  return result;
327 }
