85 TriangleTraits::Ptr Index::IndexImpl::getTriangleRTree(const mesh::Mesh &mesh)
86 {

89  }
90 
91  precice::utils::Event e("query.index.getTriangleIndexTree." + mesh.getName());
92 
93  // We first generate the values for the triangle rtree.
94  // The resulting vector is a random access range, which can be passed to the
95  // constructor of the rtree for more efficient indexing.
96  std::vector<TriangleTraits::IndexType> elements;
97  elements.reserve(mesh.triangles().size());
98  for (size_t i = 0; i < mesh.triangles().size(); ++i) {
99  auto box = bg::return_envelope<RTreeBox>(mesh.triangles()[i]);
100  elements.emplace_back(std::move(box), i);
101  }
102

104  // the best we can do.
105  impl::RTreeParameters params;
106  TriangleTraits::IndexGetter ind;
107 
108  auto tree = std::make_shared<TriangleTraits::RTree>(elements, params, ind);
109  indices.triangleRTree = std::move(tree);
