43  const std::array<double, 3> &u,
44  std::array<double, 3> v,
45  const std::array<bool, 3> & activeAxis)
46 {
47  // Substract the values and multiply out dead dimensions
48  for (unsigned int d = 0; d < v.size(); ++d) {
49  v[d] = (u[d] - v[d]) * static_cast<int>(activeAxis[d]);
50  }
51  return std::accumulate(v.begin(), v.end(), static_cast<double>(0.), [](auto &res, auto &val) { return res + val * val; });
